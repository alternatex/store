Quickstart

Embedding

<script src="src/client/store.js"></script>
<script>
// scope
(function(root){
  // ...
});
</script>

Configure Defaults

Store.config({ 
  url: "http://localhost/datastore.php", // server side handler *
  ttl: 1000 // timeout in ms
});

Create Store

var articleStore = new Store({ namespace: 'article' });

Create Object

Variant A

var article = articleStore.create({ 
  country: 'US', 
  firstname: 'Stephen', 
  lastname: 'Colbert' 
});

Variant B

var article = articleStore.create(); 
article.set('country', 'UK');
article.set('firstname', 'John'); 
article.set('lastname', 'Oliver');

Store Object

Variant A

article.update();

Variant B

articleStore.update(article);

Variant C

articleStore.update({ name: 'XXXXX', type: 'XXXXX', rank: 'XXXXX' });

Remove Object

Variant A

article.remove();

Variant B

articleStore.remove(article);

<script src="src/client/store.js"></script>
<script>

// scope
(function(root){ 

  // store `article` w/handlers 
  Store.when(article.update()).done(function(){
    // ...
  }).fail(function(error){
    // ...
  }).always(function(){
    // ...
  });

  // store `article` w/handlers & promise chain
  article.update().done(function(){
    // ...
  }).fail(function fail(){
    // ...
  }).always(function always(){
    // ...
  });

  // store `article` xxx
  var updateArticle = articleStore.update(article, function callback(){
    console.log('shalalalala');
  });

  // register handlers 
  Store.when(updateArticle).done(function(){
    // ...
  }).fail(function(){
    // ...
  }).always(function(){
    // ...
  });

  // edit `article`
  article.setData({
    id: 'fd4a722f-b2d4-0a7a-a6dd-198369e1abc5', // article.get('id'),
    country: 'US', 
    firstname: 'John', 
    lastname: 'Stewart'
  }); // or bulk em' post object initialization

  // store `article`
  article.update();  

  // remove `article`
  article.remove();

  // test
  Store.when(articleStore.list(), function(articles){
    assert(articles.length, 0);
  });

  // create some stores
  ['author', 'article', 'magazine', 'publisher'].forEach(
    function(context){
      root[context+'Datastore'] = new Store({ namespace: context });
    }
  );

  // update promises
  var updateArticles = [];

  // create 10 articles
  Store.times(10, function(count){

    // create & store
    updateArticles.push(articleStore.create({ 

      // article
      title: 'No.' + count,       
      abstract: 'Lorem Ipsum [...]',       
      text: 'Lorem Ipsum Si Amet They Say',             

      // history 
      author: 'me', 
      lastmod: new Date().getTime()
    }).update());

  });

  // when article bulk creation processed
  Store.when(updateArticles).done(function(){
    // ...
  }).fail(function(){
    // ...
  }).always(function(){
    // ...
  });

  // fetch all articles & register callback 
  Store.when(articleStore.list()).done(function(articles){

    // do stuff
    articles.forEach(function(article, index){
      console.log(index, article);
    });

  });

  // fetch article 
  var article = articleStore.get('8c0c1ff0-d0fe-38b7-376a-b0b1d53bd557');

  // remove article
  article.remove();

  // ...

})(this);
</script>

Files

<script>
  // create container
  var form = new FormData();

  // configure
  form.append('namespace', 'article');
  form.append('action', 'update');  

  // attach file
  var body = '<a id="a"><b id="b">hey!</b></a>';
  var blob = new Blob([body], { type: 'text/xml' });
  form.append('xmlfile', blob);

  // store
  var request = new XMLHttpRequest();
  request.open('POST', 'http://localhost/datastore.php');
  request.send(form);
</script>

HTML

Arbitrary Objects

<form method="POST" action="datastore.php">
  <input type="hidden" name="namespace" value="article"/>
  <input type="hidden" name="action" value="update"/>
  <input type="hidden" name="instance[id]" value="a5b1da03-c0d1-5f49-19e4-de94bb266d2e"/>
  <input type="text" name="instance[firstname]" value="firstname"/>
  <input type="text" name="instance[lastname]" value="lastname"/>
  <button data-action="update">Update</button>
  <button data-action="cancel">Cancel</button>
</form>
