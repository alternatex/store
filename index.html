Store

A simple datastore for rapid prototype development

Huh?

Example usage

<script src="src/store.js"></script>
<script>

// scope
(function(root){

  // configure
  Store.config({ 
    url: "http://localhost/datastore.php", // server side handler *
    ttl: 1000 // timeout in ms
  });

  // create entities/stores
  ['author', 'article', 'magazine', 'publisher'].forEach(
    function(context){
      root[context+'Datastore'] = new Store({ classname: context });
    }
  );

  // create 10 articles
  var i = 0;
  while(i<10 && i++) {

    // create & store
    articleStore.create({ 

      // article
      title: 'No.' + i,       
      abstract: 'Lorem Ipsum [...]',       
      text: 'Lorem Ipsum Si Amet They Say',             

      // history 
      author: 'me', 
      lastmod: new Date().getTime()
    }).update();  
  }

  // fetch all articles (10)
  articleStore.list();

  // fetch article 
  var article = articleStore.get("UUID");

  // remove article
  article.remove();

  // 

})(this);
</script>
