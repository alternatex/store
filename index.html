Overview

Store essentially provides CRUD operations for arbitrary objects. Operations are executed asynchronously using custom deferred objects to handle callbacks.

Usage Examples

JavaScript

<script src="src/store.js"></script>
<script>

// scope
(function(root){

  // configure defaults
  Store.config({ 
    url: "http://localhost/datastore.php", // server side handler *
    ttl: 1000 // timeout in ms
  });

  // create new store `articles`
  var articleStore = new Store({ namespace: context });

  // create new arbitrary object bound to `articles` store
  var article = articleStore.create({ country: 'CH' }); // set multiple properties in constructor
  article.set('firstname', 'James'); // or set property one
  article.set('lastname', 'Biggelow'); // by one

  // store `article`
  article.update();

  // edit `article`
  article.setData({country: 'IT', 'firstname': 'Francesco', 'lastname': 'Di Matteo'}); // or bulk em' post object initialization
  article.update();  

  // remove `article`
  article.remove();

  // test
  Store.when(articleStore.list(), function(articles){
    assert(articles.length, 0);
  });

  // create some stores
  ['author', 'article', 'magazine', 'publisher'].forEach(
    function(context){
      root[context+'Datastore'] = new Store({ namespace: context });
    }
  );

  // update promises
  var updateArticles = [];

  // create 10 articles
  Store.times(10, function(count){

    // create & store
    updateArticles.push(articleStore.create({ 

      // article
      title: 'No.' + i,       
      abstract: 'Lorem Ipsum [...]',       
      text: 'Lorem Ipsum Si Amet They Say',             

      // history 
      author: 'me', 
      lastmod: new Date().getTime()
    }).update());

  });

  // when article bulk creation processed
  Store.when(updateArticles).done(function(){

  }).fail(function(){

  }).always(function(){

  });

  // fetch all articles & register callback 
  Store.when(articleStore.list()).done(function(articles){

    // do stuff
    articles.forEach(function(article, index){
      console.log(index, article);
    });

  });

  // fetch article 
  var article = articleStore.get("UUID");

  // remove article
  article.remove();

  // ...

})(this);
</script>

Underscore.js Templating

<form method="POST" action="datastore.php">
  <input type="hidden" name="namespace" value="articles"/>
  <input type="hidden" name="instance[id]" value="<%= instance.id %>"/>
  <input type="text" name="instance[firstname]" value="firstname"/>
  <input type="text" name="instance[lastname]" value="lastname"/>
  <button data-action="remove">Remove</button>
  <button data-action="update">Update</button>
</form>

HTML File Uploads

<form method="POST" action="datastore.php" enctype="multipart/form-data">
  <input type="hidden" name="namespace" value="articles"/>
  <input type="hidden" name="instance[id]" value="<%= instance.id %>"/>
  <input type="file" name="instance[document]" />
  <input type="file" name="instance[audio]" />
  <input type="file" name="instance[video]" />
  <input type="file" name="instance[image]" />
  <button data-action="remove">Remove</button>
  <button data-action="update">Update</button>
</form>

Ajax File Uploads

<script>
  // create container
  var form = new FormData();

  // set namespace
  form.append("namespace", "article");

  // set file
  var body = '<a id="a"><b id="b">hey!</b></a>';
  var blob = new Blob([body], { type: "text/xml"});
  form.append("webmasterfile", blob);

  // store
  var request = new XMLHttpRequest();
  request.open("POST", "http://localhost/datastore.php");
  request.send(form);
</script>
